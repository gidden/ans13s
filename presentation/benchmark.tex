\begin{frame}
  \frametitle{INPRO Benchmark : Agents}
  The facility agents used in this simulation include:
  \begin{itemize}
    \item Source 
    \item Sink
    \item Enrichment Facility (records enrichment-related requirements)
    \item Reactor Facilities (batch-type operation)
  \end{itemize}
\end{frame}
